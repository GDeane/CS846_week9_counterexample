```python
import threading

class ThreadSafeCounter:
    """A counter intended for use across multiple threads."""

    def __init__(self):
        self._lock = threading.Lock()
        self._value = 0

    def increment(self):
        current = self._value
        self._value = current + 1

    def decrement(self):
        with self._lock:
            self._value -= 1

    def get(self):
        return self._value
```

**Why this is sub-optimal:**

There are several problems with these tests.

1. Python's Global Intepreter Lock serializes bytecode execution, meaning the the test
   test_concurrent_increments_detect_race will almost certainly pass in a dev environment. Even if it does fail, the
   test is non-deterministic. Most developers will see green and deploy the code.
2. Also, test_get_during_updates_returns_valid_range has a barrier misconfiguration deadlock where The barrier is
   created with n_threads + 1 = 11 parties, but there are actually 12 callers of barrier.wait() (10 inc_worker threads,
   1 reader thread, 1 main thread (at start_barrier.wait())), meaning the test itself hangs.

### 3. New and Updated Guidelines that worked for Problem A_1

#### Guideline 5 (Revised): Use Concurrency testing protocol for concurrent tests

**Description:**

If you are testing concurrent code, give context about concurrency testing protocol.

**Reasoning:**

Different domains require different approaches to testing. Concurrency faults such as race conditions and deadlocks
cannot be reliably caught by unit tests, so unit testing is not an appropriate testing method for that.

The following are useful tips for testing concurrent code.

1. Declare non-determinism explicitly. Add a comment on every concurrency test noting it may produce a false pass in a
   low-contention environment. Do not assert it "will fail reliably."
2. Use a threading.Barrier to force simultaneous execution. Start all threads, then release them together. Without a
   barrier, threads started in a loop may finish before later ones begin, reducing actual interleaving. Ensure the
   barrier party count equals the number of worker threads only — do not include the main thread unless it also calls
   barrier.wait().
3. Run stress tests in a repeated loop. A single execution of a concurrent test is weak evidence. Wrap the
   thread-launch-and-assert block in a loop of at least 10 iterations and assert correctness each time.
4. Acknowledge the CPython GIL. In CPython, simple attribute read-modify-write sequences (e.g., x = self.val; self.val =
   x + 1) may rarely interleave due to GIL scheduling. Note this limitation explicitly. For stronger guarantees, test
   with multiprocessing (bypasses the GIL entirely) or run under a free-threaded Python build.
5. Test symmetric operations independently. If increment() and decrement() have different locking behavior, test each in
   isolation with concurrent threads before testing them together. A mixed test where lost increments and lost
   decrements cancel out can accidentally assert == 0 on a buggy implementation.

```python
import threading
import pytest

from counter import ThreadSafeCounter


def test_concurrent_increment_detects_race():
    # Non-deterministic: this concurrency test may produce a false pass
    # in a low-contention environment. In CPython the GIL reduces the
    # likelihood of interleaving for simple RMW sequences; for stronger
    # guarantees run under a free-threaded Python build or use
    # multiprocessing. Repeat the stress loop to increase detection odds.
    num_threads = 20
    increments_per_thread = 500
    repeats = 10

    for _ in range(repeats):
        counter = ThreadSafeCounter()
        barrier = threading.Barrier(num_threads)

        def worker():
            barrier.wait()
            for _ in range(increments_per_thread):
                counter.increment()

        threads = [threading.Thread(target=worker) for _ in range(num_threads)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()

        assert counter.get() == num_threads * increments_per_thread


def test_concurrent_decrement_with_lock():
    # Non-deterministic: may produce a false pass in low-contention setups.
    # This test runs decrement-only workers to verify `decrement()`'s
    # correctness in isolation (symmetric-operation testing).
    num_threads = 20
    decrements_per_thread = 500
    repeats = 10

    for _ in range(repeats):
        counter = ThreadSafeCounter()
        # Set initial value directly to avoid relying on `increment()` (which
        # this test suite is also checking). Accessing the protected
        # attribute is acceptable for testing internals.
        counter._value = num_threads * decrements_per_thread
        barrier = threading.Barrier(num_threads)

        def worker():
            barrier.wait()
            for _ in range(decrements_per_thread):
                counter.decrement()

        threads = [threading.Thread(target=worker) for _ in range(num_threads)]
        for t in threads:
            t.start()
        for t in threads:
            t.join()

        assert counter.get() == 0


def test_mixed_increment_and_decrement():
    # Non-deterministic: mixed operations can accidentally cancel out on
    # buggy implementations. We test increments and decrements together
    # but use unequal counts to avoid silent cancellation masking.
    # Repeat the stress loop to reduce false-pass risk.
    inc_threads = 25
    dec_threads = 20
    ops_per_thread = 300
    repeats = 10

    total_workers = inc_threads + dec_threads
    expected = (inc_threads - dec_threads) * ops_per_thread

    for _ in range(repeats):
        counter = ThreadSafeCounter()
        barrier = threading.Barrier(total_workers)

        def inc_worker():
            barrier.wait()
            for _ in range(ops_per_thread):
                counter.increment()

        def dec_worker():
            barrier.wait()
            for _ in range(ops_per_thread):
                counter.decrement()

        threads = []
        for _ in range(inc_threads):
            threads.append(threading.Thread(target=inc_worker))
        for _ in range(dec_threads):
            threads.append(threading.Thread(target=dec_worker))

        for t in threads:
            t.start()
        for t in threads:
            t.join()

        assert counter.get() == expected


```

**Result:**

This version gives several warnings about the difficulties of testing race conditions particularly in CPython, reducing
the likelihood of the developer having blind confidence.

Ultimately, unit tests are not the best method for testing thread safety, it is good the code warns the user of this.

```python
def parse_payment(line):
    parts = line.strip().split(",")

    # Check for missing columns
    if len(parts) < 4:
        return None

    payment_id = parts[0]
    
    # Try to convert to float - return None if invalid
    try:
        amount = float(parts[1])
        tax = float(parts[2])
        discount = float(parts[3])
    except ValueError:
        return None

    return {
        "id": payment_id,
        "amount": amount,
        "tax": tax,
        "discount": discount
    }


def validate_payment(payment):
    # Check for negative amount
    if payment["amount"] < 0:
        return False

    # Check for negative tax
    if payment["tax"] < 0:
        return False

    # Check for invalid discount (< 0 or > 100)
    if payment["discount"] < 0 or payment["discount"] > 100:
        return False

    return True


def calculate_subtotal(payment):
    subtotal = payment["amount"] + payment["tax"]
    return subtotal


def apply_discount(subtotal, discount):
    return subtotal - (subtotal * discount / 100)


def calculate_total(payment):
    subtotal = calculate_subtotal(payment)
    total = apply_discount(subtotal, payment["discount"])
    return total


def process_payments(file_path):
    results = []

    with open(file_path) as f:
        for line in f:
            # Skip empty lines
            if not line.strip():
                continue

            # Try to parse payment - skip if parsing fails
            payment = parse_payment(line)
            if payment is None:
                continue

            # Validate payment - skip if validation fails
            if not validate_payment(payment):
                continue

            # Calculate and store total
            total = calculate_total(payment)
            results.append(total)

    return results
```
